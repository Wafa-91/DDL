CREATE TABLE Customer(
Customer-id VARCHAR(20) CONSTRAINT pk-customer PRIMARY KEY,
Customer-Name VARCHAR(20) CONSTRAINT nn-customer NOT NULL,
Customer-tel NUMBER
);


CREATE TABLE  Product(
Product-id  VARCHAR2(20)  CONSTRAINT pk-Product PRIMARY KEY,
product-Name  VARCHAR2(20) CONSTRAINT nn-Product NOT NULL,
price NUMBER  CONSTRAINT check_p CHECK (price>0)
);

CREATE TABLE  Orders(
quantity NUMBER,
total_amount NUMBER,
CONSTRAINT fk-orders-Cust FOREIGN KEY (Customer-id) REFERENCES Customer(Customer-id) ,
CONSTRAINT fk-Orders-Prod FOREIGN KEY (Product_id) REFERENCES Product(Product-id) 
);
